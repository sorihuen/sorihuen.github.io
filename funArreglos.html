Organizar todos los puntos del taller de arreglos en modo funciones:

Punto 1

Crear un vector de tipo Entero con 5 posiciones, llenarlo con información solicitada al usuario. Después de recoger toda la información, se requiere imprimir el índice de cada posición en el arreglo con su valor de la siguiente manera: [0] = 55, [1] = 99, [2] = 11, [3] = 56, [4] = 69.

Codigo

                
                    SubProceso vect ()
                        Definir indice, v Como Entero;
                        Dimension v[5];
                        v[0] <- 55;
                        v[1] <- 99;
                        v[2] <- 11;
                        v[3] <- 56;
                        v[4] <- 69;
                        Para indice <- 0 Hasta 4  Con Paso 1 Hacer
                            Escribir v[indice];
                        FinPara
                        
                    FinSubProceso

                    Proceso Vector
                        Definir v Como Entero;
                        Dimension v[5];
                        
                        vect();
                    FinProceso

Punto 2

Números pares: 2, 4, 6, 8, 10
Números impares: 1, 3, 5, 7, 9


Codigo

                
                    SubProceso numImpares(num Por Referencia, filas ,columna)
                        Definir x, i ,impar Como Entero;
                        Para i<-1 Hasta 1 Con Paso 1 Hacer
                            Escribir "Numeros Impares:" , " ", Sin Saltar;
                            Para x <- 0 Hasta 19 Con Paso 1 Hacer
                                num[i,x] <- Aleatorio(0,100);

                                    impar <- num[i,x];

                                FinSi
                            FinPara
                            Escribir " ";
                        FinPara
                        Escribir " ";
                    FinSubProceso

                    SubProceso frame_x(mat,fila,columna)
                        Para fila <- 1 Hasta 10 Con Paso 1 Hacer

                                Si fila <> 10 Entonces

                                FinSi
                            FinPara
